import { useState } from "react";
import { User } from "@/contexts/user/types";
import {
  SessionData,
  SessionSlice,
  GameScreen,
  GameDifficulty,
  Frequency,
  QuestionMode,
  Question,
  QuestionStatus,
} from "./types";
import { DEFAULT_SESSION, MAX_STEP } from "./constants";
export const useSessionSlice = (): SessionSlice => {
  const [data, setData] = useState<SessionData | null>(null);
  const patch = (partial: Partial<SessionData>) =>
    setData((prev) => (prev ? { ...prev, ...partial } : prev));
  const createSession = (user: User) =>
    setData({
      ...DEFAULT_SESSION,
      players: [{ name: user.username, isMainUser: true }],
    });
  const nextStep = () =>
    setData((prev) =>
      prev && prev.stepIndex < MAX_STEP
        ? { ...prev, stepIndex: prev.stepIndex + 1 }
        : prev,
    );
  const prevStep = () =>
    setData((prev) =>
      prev && prev.stepIndex > 1
        ? { ...prev, stepIndex: prev.stepIndex - 1 }
        : prev,
    );
  const reset = () => setData({ ...DEFAULT_SESSION, players: [] });
  const setScreen = (screen: GameScreen) => patch({ screen });
  const setTitle = (title: string) => patch({ title });
  const setFrequency = (frequency: Frequency) => patch({ frequency });
  const setDifficulty = (difficulty: GameDifficulty) => patch({ difficulty });
  const setQuestionModes = (questionModes: QuestionMode[]) =>
    patch({ questionModes });
  const setIsFreeAnswer = (isFreeAnswer: boolean) => patch({ isFreeAnswer });
  const setHasNotification = (hasNotification: boolean) =>
    patch({ hasNotification });
  const addQuestion = (question: Question) =>
    setData((prev) =>
      prev ? { ...prev, questions: [...prev.questions, question] } : prev,
    );
  const updateQuestionStatus = (id: string, status: QuestionStatus) =>
    setData((prev) => {
      if (!prev) return prev;
      const questions = prev.questions.map((q) =>
        q._id === id ? { ...q, status } : q,
      );
      return { ...prev, questions };
    });
  return {
    data,
    createSession,
    setScreen,
    setTitle,
    setFrequency,
    setDifficulty,
    setQuestionModes,
    setIsFreeAnswer,
    setHasNotification,
    addQuestion,
    updateQuestionStatus,
    nextStep,
    prevStep,
    reset,
  };
};

import { User, UserSlice, TripRecord } from "./types";
export const useUserSlice = (): UserSlice => {
  const [data, setData] = useState<User | null>(null);
  const login = (user: User) => setData(user);
  const logout = () => setData(null);
  const addTrip = (trip: TripRecord) =>
    setData((prev) =>
      prev
        ? {
            ...prev,
            history: {
              ...prev.history,
              totalTrips: prev.history.totalTrips + 1,
              totalQuestions: prev.history.totalQuestions + trip.questionsCount,
              trips: [...prev.history.trips, trip],
            },
          }
        : prev,
    );
  return { data, login, logout, addTrip };

# OnTheQuizzAgain
## Services (`src/services/`)
Chaque fichier correspond à un appel API. La convention :
- nom = méthode HTTP + ressource (`postLogin`, `getWelcome`…)
- les types `Request` et `Response` sont définis en haut du fichier
L'instance axios est configurée dans `src/services/fetch.ts` :
- `baseURL` centralisée dans `src/constants.ts`
- intercepteur request : injecte le token Bearer depuis les cookies (`js-cookie`)
- intercepteur response : retourne directement `response.data`, gère les erreurs
## Context (`src/contexts/`)
Un seul context (`AppContext`) avec un seul provider (`AppProvider`). La logique est découpée en **slices** par domaine :
- `src/contexts/user/userSlice.ts` → état et actions liés à l'utilisateur connecté (`login`, `logout`)
- `src/contexts/player/playerSlice.ts` → liste de joueurs (`add`, `remove`)
Les hooks d'accès sont exposés depuis `AppContext.ts` :
- `useUser()` → `{ data, login, logout }`
- `usePlayer()` → `{ data, add, remove }`
Ajouter un nouveau domaine = créer un nouveau dossier avec un slice, brancher dans `AppProvider`.
